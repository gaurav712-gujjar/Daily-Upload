import pandas as pd
import numpy as np

def calculate_metrics(data):
    metrics = {}
    metrics['mean'] = np.mean(data)
    metrics['std'] = np.std(data)
    metrics['sharpe'] = metrics['mean'] / metrics['std']
    return metrics

# vectorized operations
